## 2. 串口通讯子uon.txt

```cpp
// 定义输出引脚
const int outputPin12 = 12;
const int outputPin13 = 13;

// 用于存储接收到的串口数据
String incomingData = "";

// 存储引脚 12 和 13 的当前状态
bool pin12State = LOW;
bool pin13State = LOW;

void setup() {
  // 初始化串口通信，波特率为 115200
  Serial.begin(115200);

  // 将输出引脚设置为输出模式
  pinMode(outputPin12, OUTPUT);
  pinMode(outputPin13, OUTPUT);

  // 初始化输出引脚的电平为低电平
  digitalWrite(outputPin12, pin12State);
  digitalWrite(outputPin13, pin13State);

  Serial.println("Serial communication started. Waiting for input...");
}

void loop() {
  // 检查是否有串口数据可读
  if (Serial.available() > 0) {
    // 读取串口数据直到换行符（'\n'）
    incomingData = Serial.readStringUntil('\n');
    incomingData.trim(); // 去除字符串首尾的空白字符

    // 检查接收到的数据是否为 "12"
    if (incomingData == "12") {
      // 改变引脚 12 的状态
      pin12State = !pin12State;
      digitalWrite(outputPin12, pin12State);
      Serial.print("Received '12'. Pin 12 state changed to ");
      Serial.println(pin12State ? "HIGH" : "LOW");
    } 
    // 检查接收到的数据是否为 "13"
    else if (incomingData == "13") {
      // 改变引脚 13 的状态
      pin13State = !pin13State;
      digitalWrite(outputPin13, pin13State);
      Serial.print("Received '13'. Pin 13 state changed to ");
      Serial.println(pin13State ? "HIGH" : "LOW");
    } 
    else {
      Serial.print("Received unknown data: ");
      Serial.println(incomingData);
    }
  }
}
```

## 5. 单机，双击，长按控制3个灯.txt

```cpp
// 定义开关引脚
const int switchPin = 3;
// 定义第一个灯的引脚
const int ledPin = 13;
// 定义长按控制的灯的引脚
const int longPressLedPin = 12;
// 定义双击控制的灯的引脚
const int doubleClickLedPin = 11;

// 用于存储开关的上一个状态
int lastSwitchState = LOW;
// 用于存储开关的当前状态
int currentSwitchState;
// 用于标记开关是否被按下
bool switchPressed = false;
// 用于标记第一个灯的当前状态，初始为熄灭状态
bool ledOn = false;
// 用于标记长按控制的灯的当前状态，初始为熄灭状态
bool longPressLedOn = false;
// 用于标记双击控制的灯是否闪烁，初始为不闪烁
bool isDoubleClickLedBlinking = false;
// 记录开关按下的时间
unsigned long pressStartTime = 0;
// 记录上一次按下开关的时间
unsigned long lastPressTime = 0;
// 长按的时间阈值（毫秒）
const unsigned long longPressThreshold = 1000;
// 双击的时间阈值（毫秒）
const unsigned long doubleClickThreshold = 300;
// 记录按下次数
int pressCount = 0;
// 闪烁间隔时间（毫秒）
const unsigned long blinkInterval = 200;
// 记录上次闪烁的时间
unsigned long lastBlinkTime = 0;
// 标记灯当前的闪烁状态
bool blinkState = false;

void setup() {
  // 将开关引脚设置为输入模式
  pinMode(switchPin, INPUT);
  // 将第一个灯的引脚设置为输出模式
  pinMode(ledPin, OUTPUT);
  // 将长按控制的灯的引脚设置为输出模式
  pinMode(longPressLedPin, OUTPUT);
  // 将双击控制的灯的引脚设置为输出模式
  pinMode(doubleClickLedPin, OUTPUT);
  // 初始化三个灯为熄灭状态
  digitalWrite(ledPin, LOW);
  digitalWrite(longPressLedPin, LOW);
  digitalWrite(doubleClickLedPin, LOW);
}

void loop() {
  // 读取开关的当前状态
  currentSwitchState = digitalRead(switchPin);

  // 检测开关是否被按下
  if (currentSwitchState == HIGH && lastSwitchState == LOW) {
    switchPressed = true;
    // 记录按下的时间
    pressStartTime = millis();

    // 检查是否可能是双击
    if (millis() - lastPressTime <= doubleClickThreshold) {
      pressCount++;
    } else {
      pressCount = 1;
    }
    lastPressTime = millis();
  }

  // 检测开关是否处于按下状态
  if (currentSwitchState == HIGH) {
    // 检查是否达到长按时间阈值
    if (millis() - pressStartTime >= longPressThreshold) {
      // 长按操作，切换长按控制的灯的状态
      longPressLedOn = !longPressLedOn;
      digitalWrite(longPressLedPin, longPressLedOn ? HIGH : LOW);
      // 重置开关按下标记和按下时间
      switchPressed = false;
      pressStartTime = 0;
      pressCount = 0;
      // 等待开关松开
      while (digitalRead(switchPin) == HIGH);
    }
  }

  // 检测开关是否被松开
  if (currentSwitchState == LOW && lastSwitchState == HIGH && switchPressed) {
    // 检查是否是双击
    if (pressCount == 2) {
      // 双击操作，切换双击控制的灯的闪烁状态
      isDoubleClickLedBlinking = !isDoubleClickLedBlinking;
      if (!isDoubleClickLedBlinking) {
        digitalWrite(doubleClickLedPin, LOW);
      }
      pressCount = 0;
    } else if (pressCount == 1 && millis() - lastPressTime > doubleClickThreshold) {
      // 明确是单次按下，切换第一个灯的状态
      ledOn = !ledOn;
      digitalWrite(ledPin, ledOn ? HIGH : LOW);
    }

    switchPressed = false;
  }

  // 处理双击控制的灯的闪烁逻辑
  if (isDoubleClickLedBlinking) {
    unsigned long currentTime = millis();
    if (currentTime - lastBlinkTime >= blinkInterval) {
      lastBlinkTime = currentTime;
      blinkState = !blinkState;
      digitalWrite(doubleClickLedPin, blinkState ? HIGH : LOW);
    }
  }

  // 更新上一个开关状态
  lastSwitchState = currentSwitchState;

  // 可以添加一些延迟以减少 CPU 负载
  delay(10);
}
```

## 6. 圆形按钮开关单双击检测.txt

```cpp
const int switchPin = 0;  // 开关连接到IO0引脚

// 状态变量
int lastSteadyState = HIGH;  // 上一次稳定的状态
int lastFlickerableState = HIGH;  // 上一次可能抖动的状态
int currentState;  // 当前状态

// 时间变量
unsigned long lastDebounceTime = 0;  // 上一次去抖时间
unsigned long debounceDelay = 50;    // 去抖延迟(毫秒)

// 单击双击检测
unsigned long firstClickTime = 0;    // 第一次点击时间
int clickCount = 0;                  // 点击计数
const unsigned long doubleClickDelay = 500;  // 双击最大间隔(毫秒)

void setup() {
  pinMode(switchPin, INPUT_PULLUP);  // 启用内部上拉电阻
  Serial.begin(115200);
  Serial.println("Switch test started");
}

void loop() {
  // 读取当前引脚状态
  currentState = digitalRead(switchPin);

  // 检测状态是否发生变化
  if (currentState != lastFlickerableState) {
    lastDebounceTime = millis();  // 重置去抖计时器
    lastFlickerableState = currentState;  // 保存当前状态
  }

  // 如果状态稳定超过去抖延迟
  if ((millis() - lastDebounceTime) > debounceDelay) {
    // 如果状态确实改变
    if (lastSteadyState != currentState) {
      lastSteadyState = currentState;
      
      // 检测到开关释放(从低到高的变化)
      if (currentState == HIGH) {
        clickCount++;  // 增加点击计数
        Serial.print("Click detected! Count: ");
        Serial.println(clickCount);
        
        // 如果是第一次点击，记录时间
        if (clickCount == 1) {
          firstClickTime = millis();
        }
      }
    }
  }

  // 检查双击间隔是否已过
  if (clickCount > 0 && (millis() - firstClickTime) > doubleClickDelay) {
    // 判断是单击还是双击
    if (clickCount == 1) {
      Serial.println("on");  // 单击发送"on"
    } else if (clickCount == 2) {
      Serial.println("light");  // 双击发送"light"
    } else {
      Serial.print("Multiple clicks: ");  // 多击情况
      Serial.println(clickCount);
    }
    
    // 重置点击计数
    clickCount = 0;
  }
}
```

```cpp
    if ((millis() - lastDebounceTime) > debounceDelay) {
        if (reading != switchState) {
            switchState = reading;
            if (switchState == LOW) {
                Serial.println("Emergency stop triggered!");
                // 停止电机脉冲输出，实现急停

        // 去除接收到的字符串首尾的空白字符

            Serial.println("Received'm1zz'. Rotating motor 1 clockwise (正转) one revolution...");
            // 调用 rotateMotor 函数控制电机1正转一圈
            rotateMotor(true, false, stepsPerRevolution, pulPin1, dirPin1, pulPin2, dirPin2);

// 参数：
// clockwise1 表示电机1的旋转方向，true 为正转，false 为反转
// clockwise2 表示电机2的旋转方向，true 为正转，false 为反转

// pulPin1 为电机1的脉冲引脚
// dirPin1 为电机1的方向引脚
// pulPin2 为电机2的脉冲引脚
// dirPin2 为电机2的方向引脚
void rotateMotor(bool clockwise1, bool clockwise2, int steps, int pulPin1, int dirPin1, int pulPin2, int dirPin2) {
    // 根据 clockwise1 的值设置电机1方向引脚的电平，控制电机1旋转方向
    digitalWrite(dirPin1, clockwise1 ? HIGH : LOW);
    // 根据 clockwise2 的值设置电机2方向引脚的电平，控制电机2旋转方向
    digitalWrite(dirPin2, clockwise2 ? HIGH : LOW);

    // 循环发送指定数量的脉冲，使两个电机同时旋转相应步数
    for (int i = 0; i < steps; i++) {
        int currentSwitchState = digitalRead(switchPin);
        if (currentSwitchState != lastSwitchState) {
            lastDebounceTime = millis();
        }
        if ((millis() - lastDebounceTime) > debounceDelay) {
            if (currentSwitchState != switchState) {
                switchState = currentSwitchState;
                if (switchState == LOW) {
                    Serial.println("Emergency stop triggered during rotation!");
                    // 停止电机脉冲输出，实现急停
                    digitalWrite(pulPin1, LOW);
                    digitalWrite(pulPin2, LOW);
                    return;
                }
            }
        }
        lastSwitchState = currentSwitchState;

        // 将电机1和电机2的脉冲引脚同时设置为高电平
        digitalWrite(pulPin1, HIGH);
        digitalWrite(pulPin2, HIGH);
        // 延迟 500 微秒，控制脉冲频率，进而调整电机转速
        delayMicroseconds(500);
        // 将电机1和电机2的脉冲引脚同时设置为低电平
        digitalWrite(pulPin1, LOW);
        digitalWrite(pulPin2, LOW);
        // 延迟 500 微秒
        delayMicroseconds(500);
    }
}
```

## 16. 舵机0到180来回摆动.txt

```cpp
#include <Servo.h>

//代码实现了舵机先从任意初始位置回到 0 度，再转动到 90 度，之后以 15 毫秒为间隔、1 度为步长在 -90 度到 90 度间来回摆动且摆到头才改变方向循环的功能。 
// 定义舵机控制引脚
const int servoPin = 3;
// 创建舵机对象
Servo myServo;
// 舵机当前角度
int angle;
// 每次转动的角度步长
const int step = 1;
// 转动间隔时间（毫秒），可调整来改变摆动速度
const unsigned long interval = 15;
// 记录上次执行时间
unsigned long lastTime = 0;
// 标记摆动方向，true 表示正向（角度增加），false 表示反向（角度减小）
bool direction = true;
// 标记是否已经完成初始归零到 0 度
bool initializedToZero = false;
// 标记是否已经转动到最大角度（90 度）
bool reachedMax = false;

void setup() {
  // 初始化舵机连接
  myServo.attach(servoPin);
  // 假设舵机初始位置未知，这里先读取当前舵机位置
  angle = myServo.read() - 90;
}

void loop() {
  unsigned long currentTime = millis();

  // 检查是否达到转动间隔时间
  if (currentTime - lastTime >= interval) {
    lastTime = currentTime;

    if (!initializedToZero) {
      // 还未完成初始归零到 0 度
      if (angle < 0) {
        angle += step;
      } else if (angle > 0) {
        angle -= step;
      }
      myServo.write(angle + 90);

      if (angle == 0) {
        // 到达 0 度，完成初始化
        initializedToZero = true;
      }
    } else if (!reachedMax) {
      // 已归零到 0 度，但还未转动到最大角度（90 度）
      if (angle < 90) {
        angle += step;
        myServo.write(angle + 90);
      } else {
        // 到达 90 度
        reachedMax = true;
        direction = false; // 开始反向摆动
      }
    } else {
      // 已经转动到最大角度，开始来回摆动
      if (direction) {
        // 正向摆动，角度增加
        if (angle < 90) {
          angle += step;
          myServo.write(angle + 90);
        } else {
          // 到达 90 度，改变摆动方向
          direction = false;
        }
      } else {
        // 反向摆动，角度减小
        if (angle > -90) {
          angle -= step;
          myServo.write(angle + 90);
        } else {
          // 到达 -90 度，改变摆动方向
          direction = true;
        }
      }
    }
  }
}
```

```cpp
#include <Servo.h>

// --------------------------

const int maxPulse = 2500;       // 最大脉冲宽度（μs），对应最大角度
const int moveDelay = 15;        // 角度切换间隔（毫秒），值越小转动越快

// 创建舵机对象
Servo myServo;

void setup() {
  // 初始化串口通信（用于调试）
  Serial.begin(115200);
  
  // 将舵机连接到指定引脚，并设置脉冲范围
  myServo.attach(servoPin, minPulse, maxPulse);
  
  // 初始化舵机到最小角度位置
  myServo.write(minAngle);
  Serial.print("初始化到");
  Serial.print(minAngle);
  Serial.println("度位置");
  delay(pauseTime); // 等待舵机到达初始位置

  Serial.print("开始从");

  Serial.print("度转到");
  Serial.print(maxAngle);
  Serial.println("度");
  
  for (int angle = minAngle; angle <= maxAngle; angle++) {
    myServo.write(angle);  // 设置舵机角度
    Serial.print("当前角度: ");
    Serial.println(angle);
    delay(moveDelay);      // 控制转动速度

  Serial.print("已到达");
  Serial.print(maxAngle);

  Serial.print("开始从");
  Serial.print(maxAngle);
  Serial.print("度转回");
  Serial.print(minAngle);

  for (int angle = maxAngle; angle >= minAngle; angle--) {
    myServo.write(angle);  // 设置舵机角度
```

```cpp
// 定义输出引脚
const int outputPins[] = {4, 5, 6, 7, 8, 9, 10, 11};

int currentState = 0;

void setup() {
    // 初始化串口通信，波特率为 9600
    Serial.begin(9600);

    // 将输出引脚设置为输出模式
    for (int i = 0; i < numPins; i++) {
        pinMode(outputPins[i], OUTPUT);
        digitalWrite(outputPins[i], LOW);
    }

    Serial.println("Serial communication started. Waiting for input...");
}

void loop() {
    // 检查是否有串口数据可读
    if (Serial.available() > 0) {
        // 读取串口数据直到换行符（'\n'）
        incomingData = Serial.readStringUntil('\n');
        incomingData.trim(); // 去除字符串首尾的空白字符

        if (incomingData == "upgo") {
            if (currentState == 1) {
                // 再次输入 upgo，恢复到初始状态
                for (int i = 0; i < numPins; i++) {
                    digitalWrite(outputPins[i], LOW);

                currentState = 0;
```
