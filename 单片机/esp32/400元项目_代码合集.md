## 1. 基础 - MPU6050 X轴角度读取

> 文件名：`arduino,mpu6050x轴角度.txt`

```cpp
#include <Wire.h>

const int MPU_addr = 0x68; // I2C address of the MPU-6050
int16_t AcX, AcZ;

void setup() {
  Wire.begin();
  Wire.beginTransmission(MPU_addr);
  Wire.write(0x6B);  // PWR_MGMT_1 register
  Wire.write(0);     // set to zero (wakes up the MPU-6050)
  Wire.endTransmission(true);
  Serial.begin(9600);
}

void loop() {
  Wire.beginTransmission(MPU_addr);
  Wire.write(0x3B);  // starting with register 0x3B (ACCEL_XOUT_H)
  Wire.endTransmission(false);
  Wire.requestFrom(MPU_addr, 6, true); // request a total of 6 registers (X, Y, Z)

  AcX = Wire.read() << 8 | Wire.read(); // 0x3B (ACCEL_XOUT_H) & 0x3C (ACCEL_XOUT_L)
  Wire.read(); Wire.read(); // skip Y axis
  AcZ = Wire.read() << 8 | Wire.read(); // 0x43 (ACCEL_ZOUT_H) & 0x44 (ACCEL_ZOUT_L)

  // Convert raw values to g
  float accelXG = (float)AcX / 16384.0;
  float accelZG = (float)AcZ / 16384.0;

  // Calculate the angle in degrees
  float angleX = atan2(accelXG, accelZG) * (180 / PI);

  // Convert the angle to the 0 - 360 degrees range
  if (angleX < 0) {
    angleX += 360;
  }

  Serial.print("当前（");
  Serial.print(angleX);
  Serial.println("）°");
  delay(100);
}
```

## 2. 基础 - MPU6050 陀螺仪旋转速度

> 文件名：`mpu6050旋转速度.txt`

```cpp
#include <Wire.h>

const int MPU_addr = 0x68; // I2C address of the MPU - 6050
int16_t AcX, AcZ;
int16_t GyY;

void setup() {
  Wire.begin();
  Wire.beginTransmission(MPU_addr);
  Wire.write(0x6B);  // PWR_MGMT_1 register
  Wire.write(0);     // set to zero (wakes up the MPU - 6050)
  Wire.endTransmission(true);
  Serial.begin(9600);
}

void loop() {
  // Read accelerometer data
  Wire.beginTransmission(MPU_addr);
  Wire.write(0x3B);  // starting with register 0x3B (ACCEL_XOUT_H)
  Wire.endTransmission(false);
  Wire.requestFrom(MPU_addr, 6, true); // request a total of 6 registers (X, Y, Z)

  AcX = Wire.read() << 8 | Wire.read(); // 0x3B (ACCEL_XOUT_H) & 0x3C (ACCEL_XOUT_L)
  Wire.read(); Wire.read(); // skip Y axis
  AcZ = Wire.read() << 8 | Wire.read(); // 0x43 (ACCEL_ZOUT_H) & 0x44 (ACCEL_ZOUT_L)

  // Convert raw values to g
  float accelXG = (float)AcX / 16384.0;
  float accelZG = (float)AcZ / 16384.0;

  // Calculate the angle in degrees
  float angleX = atan2(accelXG, accelZG) * (180 / PI);

  // Convert the angle to the 0 - 360 degrees range
  if (angleX < 0) {
    angleX += 360;
  }

  // Read gyroscope data
  Wire.beginTransmission(MPU_addr);
  Wire.write(0x45);  // starting with register 0x45 (GYRO_YOUT_H)
  Wire.endTransmission(false);
  Wire.requestFrom(MPU_addr, 2, true); // request 2 registers for Y axis gyro data

  GyY = Wire.read() << 8 | Wire.read(); // 0x45 (GYRO_YOUT_H) & 0x46 (GYRO_YOUT_L)

  // Convert raw gyro value to degrees per second
  float gyroYDegPerSec = (float)GyY / 131.0;

  // Make the gyro speed positive
  gyroYDegPerSec = abs(gyroYDegPerSec);

  Serial.print("当前角度（");
  Serial.print(angleX);
  Serial.print("）°，y轴旋转速度（");
  Serial.print(gyroYDegPerSec);
  Serial.println("）°/s");
  delay(100);
}
```

## 4. 基础 - 步进电机基本旋转

> 文件名：`电机旋转.txt`

```cpp
const int dirPin = 19;
const int pulPin = 18;
// 步进电机每转所需的脉冲数
const int stepsPerRevolution = 400; 
// 旋转180度所需的脉冲数
const int stepsFor180Degrees = stepsPerRevolution / 2; 
// 慢速度阶段的脉冲间隔时间（微秒）
const int slowDelay = 1600; 
// 快速度阶段的脉冲间隔时间（微秒）
const int fastDelay = 800; 
// 慢速度阶段的步数
const int slowSteps = stepsFor180Degrees / 4; 

// 封装步进电机旋转 180 度的函数
void rotateMotor180Degrees() {
    // 设置旋转方向
    digitalWrite(dirPin, HIGH);

    // 慢速度阶段
    for (int i = 0; i < slowSteps; i++) {
        digitalWrite(pulPin, HIGH);
        delayMicroseconds(slowDelay);
        digitalWrite(pulPin, LOW);
        delayMicroseconds(slowDelay);
    }

    // 快速度阶段
    for (int i = 0; i < stepsFor180Degrees - 2 * slowSteps; i++) {
        digitalWrite(pulPin, HIGH);
        delayMicroseconds(fastDelay);
        digitalWrite(pulPin, LOW);
        delayMicroseconds(fastDelay);
    }

    // 慢速度阶段
    for (int i = 0; i < slowSteps; i++) {
        digitalWrite(pulPin, HIGH);
        delayMicroseconds(slowDelay);
        digitalWrite(pulPin, LOW);
        delayMicroseconds(slowDelay);
    }
}

void setup() {
    // 初始化串口通信
    Serial.begin(115200);
    // 设置引脚为输出模式
    pinMode(dirPin, OUTPUT);
    pinMode(pulPin, OUTPUT);
}

void loop() {
    if (Serial.available() > 0) {
        int incomingByte = Serial.read();
        if (incomingByte == '1') {
            rotateMotor180Degrees();
        }
    }
}
```

```cpp
#define INTERPOLATION 10    // 插值倍数（越大曲线越平滑，建议5-20）
float gyroHistory[NUM_POINTS] = {0}; // 历史速度缓冲区
int dataCount = 0;                   // 当前数据点数

int axisY, axisLengthY; // 坐标系Y轴参数

// ====================== 自定义插值函数 ======================
float myLerp(float a, float b, float t) {
  return a + t * (b - a); // 线性插值公式：a + t*(b-a)
}

// ====================== 初始化函数 ======================
void setup() {
  // 初始化I2C总线（ESP32默认引脚：SDA=21，SCL=22）
  Wire.begin(21, 22);
  
  // 初始化OLED屏幕
  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println(F("SSD1306初始化失败！"));
    while (1); // 初始化失败则挂起
  }
  display.clearDisplay();          // 清屏
  display.setTextSize(1);          // 设置字体大小
  display.setTextColor(SSD1306_WHITE); // 设置字体颜色
  
  // 初始化MPU-6050传感器
  Wire.beginTransmission(MPU_addr);
  Wire.write(0x6B); // 电源管理寄存器1（PWR_MGMT_1）
  Wire.write(0x00); // 设为0，唤醒传感器
  Wire.endTransmission(true);
  
  Serial.begin(115200); // 初始化串口通信
}

// ====================== 主循环函数 ======================
void loop() {
  readAccelData();   // 读取加速度计数据
  readGyroData();    // 读取陀螺仪数据
  
  // 更新历史数据缓冲区
  if (dataCount < NUM_POINTS) {
    gyroHistory[dataCount++] = abs((float)GyY / 131.0); // 转换为°/s并取绝对值
  } else {
    // 使用内存移动优化数据移位（比循环更快）
    memmove(gyroHistory, gyroHistory + 1, (NUM_POINTS - 1) * sizeof(float));
    gyroHistory[NUM_POINTS - 1] = abs((float)GyY / 131.0);
  }

  // ====================== 显示角度信息 ======================
  display.clearDisplay(); // 清屏
  
  // 绘制汉字"当前"
  display.drawBitmap(0, 0, font_dang, 16, 16, SSD1306_WHITE);
  display.drawBitmap(16, 0, font_qian, 16, 16, SSD1306_WHITE);
  
  // 显示角度数值（保留2位小数）
  char angleStr[8];
  dtostrf(calculateAngle(), 6, 2, angleStr); // 转换为字符串
  display.setCursor(32, 5);                   // 计算显示位置（32=16*2）
  display.print(angleStr);                    // 显示数值
  
  // 绘制单位"°"
  int numWidth = getStringLength(angleStr) * 5; // 数字宽度（每个字符5像素）
  display.drawBitmap(32 + numWidth, 0, font_du, 16, 16, SSD1306_WHITE);

  // ====================== 绘制坐标系和曲线 ======================
  drawCoordinateSystem(); // 绘制坐标轴
  drawGyroCurve();        // 绘制陀螺仪速度曲线
  
  display.display(); // 刷新屏幕显示
  delay(100);        // 延时100ms（控制刷新频率）
}

// ====================== 加速度计数据读取 ======================
void readAccelData() {
  Wire.beginTransmission(MPU_addr);
  Wire.write(0x3B); // 加速度计X轴高位寄存器（0x3B-0x3C为X轴，0x3D-0x3E为Y轴，0x3F-0x40为Z轴）
  Wire.endTransmission(false);
  Wire.requestFrom(MPU_addr, 6, true); // 一次性读取6字节（X、Y、Z轴）
  
  // 解析X轴和Z轴数据（跳过Y轴）
  AcX = Wire.read() << 8 | Wire.read(); // X轴：高位+低位
  Wire.read(); Wire.read(); // 跳过Y轴
  AcZ = Wire.read() << 8 | Wire.read(); // Z轴：高位+低位
}

// ====================== 陀螺仪数据读取 ======================
void readGyroData() {
  Wire.beginTransmission(MPU_addr);
  Wire.write(0x45); // 陀螺仪Y轴高位寄存器（0x45-0x46为Y轴）
  Wire.endTransmission(false);
  Wire.requestFrom(MPU_addr, 2, true); // 读取2字节Y轴数据
  
  if (Wire.available() >= 2) {
    GyY = Wire.read() << 8 | Wire.read(); // 组合高低位数据
  } else {
    GyY = 0; // 数据不足时设为0
  }
}

// ====================== 角度计算函数 ======================
float calculateAngle() {
  float gx = (float)AcX / 16384.0; // 转换为g值（±2g量程）
  float gz = (float)AcZ / 16384.0;
  
  // 使用atan2计算角度（弧度转角度）
  float angle = atan2(gx, gz) * RAD_TO_DEG;
  
  // 角度归一化（0-360°）
  return (angle < 0) ? angle + 360 : angle;
}

// ====================== 绘制坐标系 ======================
void drawCoordinateSystem() {
  axisY = SCREEN_HEIGHT - 1;           // Y轴底部位置（屏幕最下方）
  axisLengthY = Y_AXIS_RANGE;          // Y轴显示高度
  
  // 绘制X轴（水平轴）
  display.drawLine(0, axisY, SCREEN_WIDTH - 1, axisY, SSD1306_WHITE);
  
  // 绘制Y轴（垂直轴）
  display.drawLine(0, axisY, 0, axisY - axisLengthY, SSD1306_WHITE);
  
  // 绘制Y轴箭头（三角形）
  display.drawTriangle(
    0, axisY - axisLengthY,            // 顶点
    -2, axisY - axisLengthY + 5,       // 左箭头
    +2, axisY - axisLengthY + 5,       // 右箭头
    SSD1306_WHITE
  );
  
  // 标注Y轴刻度（0和200）
  display.setTextSize(0.6);            // 设置小字体
  display.setCursor(2, axisY - axisLengthY - 4); // 顶部刻度200

  display.setCursor(2, axisY + 2);      // 底部刻度0
  display.print("0");
}

// ====================== 绘制陀螺仪曲线 ======================
void drawGyroCurve() {
  float xStep = (SCREEN_WIDTH - 1) / (NUM_POINTS - 1); // 每个原始点的X轴间隔
  float scale = axisLengthY / 200.0f;                  // 数值到像素的缩放因子（200°/s对应满刻度）
  
  // 预计算所有插值点（原始点间插入INTERPOLATION个点）
  float interpolatedPoints[(NUM_POINTS - 1) * INTERPOLATION + 1];
  
  for (int i = 0; i < NUM_POINTS; i++) {
    if (i == NUM_POINTS - 1) {
      interpolatedPoints[i * INTERPOLATION] = gyroHistory[i]; // 最后一个原始点
    } else {
      for (int j = 0; j < INTERPOLATION; j++) {
        float t = j / (float)INTERPOLATION; // 插值比例（0到1）
        int index = i * INTERPOLATION + j;
        interpolatedPoints[index] = myLerp(
          gyroHistory[i], 
          gyroHistory[i + 1], 
          t
        );
      }
    }
  }
  
  // 绘制插值后的曲线
  for (int i = 1; i < (NUM_POINTS - 1) * INTERPOLATION + 1; i++) {
    float x1 = (i - 1) * xStep / INTERPOLATION; // 前一点X坐标
    float x2 = i * xStep / INTERPOLATION;         // 当前点X坐标
    float y1 = axisY - interpolatedPoints[i - 1] * scale; // 前一点Y坐标
    float y2 = axisY - interpolatedPoints[i] * scale;     // 当前点Y坐标
    
    // 边界限制（防止越界）
    y1 = constrain(y1, axisY - axisLengthY, axisY);
    y2 = constrain(y2, axisY - axisLengthY, axisY);
    
    // 绘制线段
    display.drawLine(x1, y1, x2, y2, SSD1306_WHITE);
  }
}
```

```cpp
float minY = -1;         // Y轴最小值
float maxY = 1;          // Y轴最大值

bool drawingTriggered = false;
int currentX = 0;

    return sin(x);

const int dirPin = 19;
const int pulPin = 18;
// 步进电机每转所需的脉冲数
const int stepsPerRevolution = 400; 
// 旋转180度所需的脉冲数
const int stepsFor180Degrees = stepsPerRevolution / 2; 
// 慢速度阶段的脉冲间隔时间（微秒）
const int slowDelay = 1600; 
// 快速度阶段的脉冲间隔时间（微秒）
const int fastDelay = 800; 
// 慢速度阶段的步数
const int slowSteps = stepsFor180Degrees / 4; 

// 封装步进电机旋转 180 度的函数
void rotateMotor180Degrees() {
    // 设置旋转方向
    digitalWrite(dirPin, HIGH);

    // 慢速度阶段
    for (int i = 0; i < slowSteps; i++) {
        digitalWrite(pulPin, HIGH);
        delayMicroseconds(slowDelay);
        digitalWrite(pulPin, LOW);
        delayMicroseconds(slowDelay);
    }

    // 快速度阶段
    for (int i = 0; i < stepsFor180Degrees - 2 * slowSteps; i++) {
        digitalWrite(pulPin, HIGH);
        delayMicroseconds(fastDelay);
        digitalWrite(pulPin, LOW);
        delayMicroseconds(fastDelay);
    }

    // 慢速度阶段
    for (int i = 0; i < slowSteps; i++) {
        digitalWrite(pulPin, HIGH);
        delayMicroseconds(slowDelay);
        digitalWrite(pulPin, LOW);
        delayMicroseconds(slowDelay);
    }
}

void setup() {
    // 初始化串口通信
    Serial.begin(115200);

    // 初始化OLED
    display.begin(SSD1306_SWITCHCAPVCC, I2C_ADDR);
    display.clearDisplay();
    
    // 绘制坐标轴（原点在屏幕左下角，对应数学坐标 (-π/2, -1)）
    drawAxes();
    display.display();

    // 设置步进电机引脚为输出模式
    pinMode(dirPin, OUTPUT);
    pinMode(pulPin, OUTPUT);
}

void loop() {
    if (Serial.available() > 0) {
        char cmd = Serial.read();
        if (cmd == '1') {        // 开始绘制正弦曲线并旋转电机
            drawingTriggered = true;
            currentX = 0;
            rotateMotor180Degrees();
        } else if (cmd == '0') { // 立即清除图像
            clearGraph();
        }
    }

    if (drawingTriggered && currentX < SCREEN_WIDTH) {
        drawNextFrame(); // 绘制下一帧
    }
}

// 绘制坐标轴
void drawAxes() {
    int originY = SCREEN_HEIGHT - 1; // 原点Y坐标（屏幕底部）
    
    // X轴（水平，底部）
    display.drawLine(0, originY, SCREEN_WIDTH-1, originY, WHITE);
    
    // Y轴（垂直，左侧）
    display.drawLine(0, 0, 0, SCREEN_HEIGHT-1, WHITE);
}

// 清除图像（保留坐标轴）
void clearGraph() {
    drawingTriggered = false;
    display.clearDisplay(); // 清空像素
    drawAxes();             // 重绘坐标轴
    display.display();      // 立即刷新
    currentX = 0;
}

// 绘制下一帧动画
void drawNextFrame() {
    int originY = SCREEN_HEIGHT - 1; // 原点在屏幕底部
    
    // 计算当前x对应的数学坐标
    float x = minX + (maxX - minX) * (currentX / (float)SCREEN_WIDTH);
    float y = sineFunction(x);
    
    // 坐标映射：将数学坐标转换为屏幕像素
    int pixelX = currentX;
    int pixelY = originY - static_cast<int>((y - minY) * (SCREEN_HEIGHT-1) / (maxY - minY));
    
    // 绘制像素
    display.drawPixel(pixelX, pixelY, WHITE);
    
    // 刷新当前行（优化显示速度）
    display.ssd1306_command(SSD1306_PAGEADDR);
    display.ssd1306_command(0); // 起始页
    display.ssd1306_command(7); // 结束页
    display.display();
    
    currentX++;
    delay(3); // 总时间 128*3=384ms < 480ms
```
```cpp
}

// 最短路径控制函数（核心修正）
void moveToAngle(float targetAngle) {
  // 角度归一化（0-360°）
  targetAngle = fmod(targetAngle, 360.0);
  if (targetAngle < 0) targetAngle += 360.0;

  // 计算角度差（考虑环形特性）
  float deltaAngle = targetAngle - currentAngle;
  
  // 转换为最短路径（-180°~180°）
  if (deltaAngle > 180) deltaAngle -= 360;
  if (deltaAngle < -180) deltaAngle += 360;

  // 计算需要移动的步数
  int steps = round(deltaAngle * stepsPerDegree);
  if (steps == 0) return; // 无需移动

  // 确定转动方向
  int direction = (steps > 0) ? HIGH : LOW; // HIGH=顺时针，LOW=逆时针
  digitalWrite(dirPin, direction);

  // 计算实际步数（取绝对值）
  int stepCount = abs(steps);

  // 加减速控制（梯形速度曲线）
  const int maxSpeed = 200; // 最大速度（步/秒）
  const int acceleration = 50; // 加速度（步/秒²）
  int speed = 0;
  int stepDelay = 1000; // 初始延迟（微秒）

  for (int i = 0; i < stepCount; i++) {
    // 加速阶段
    if (speed < maxSpeed) {
      speed = min(speed + acceleration, maxSpeed);
    }
    // 减速阶段
    else if (i > stepCount - maxSpeed / acceleration) {
      speed = max(speed - acceleration, 0);
    }

    stepDelay = (int)(1000000.0 / speed); // 转换为微秒
    digitalWrite(pulPin, HIGH);
    delayMicroseconds(stepDelay / 2);
    digitalWrite(pulPin, LOW);
    delayMicroseconds(stepDelay / 2);

    // 更新当前位置并归一化（防止溢出）
    currentPosition += (direction == HIGH) ? 1 : -1;
    currentPosition = fmod(currentPosition, fullRevolution);
    if (currentPosition < 0) currentPosition += fullRevolution;
  }

  // 更新当前角度值
  currentAngle = targetAngle;
}

  Wire.endTransmission(false);

  Wire.endTransmission(false);
  Wire.requestFrom(MPU_addr, 2, true); // 读取Y轴数据
  
  if (Wire.available() >= 2) {
    GyY = Wire.read() << 8 | Wire.read(); // 组合高低位数据

  return angle;

    memmove(gyroHistory, gyroHistory + 1, (NUM_POINTS - 1) * sizeof(float));
    gyroHistory[NUM_POINTS - 1] = abs((float)GyY / 131.0);
  }
}

    SSD1306_WHITE
  );

  display.print("200");
  
  display.setCursor(2, axisY + 2);
  display.print("0");
}

// ====================== 绘制陀螺仪曲线 ======================
void drawGyroCurve() {
  float xStep = (SCREEN_WIDTH - 1) / (NUM_POINTS - 1); // 每个原始点的X轴间隔
  float scale = axisLengthY / 200.0f;                  // 数值到像素的缩放因子
  
  // 预计算插值点
  float interpolatedPoints[(NUM_POINTS - 1) * INTERPOLATION + 1];
  
  for (int i = 0; i < NUM_POINTS; i++) {
    if (i == NUM_POINTS - 1) {
      interpolatedPoints[i * INTERPOLATION] = gyroHistory[i];
    } else {
      for (int j = 0; j < INTERPOLATION; j++) {
        float t = j / (float)INTERPOLATION;
        int index = i * INTERPOLATION + j;
        interpolatedPoints[index] = myLerp(gyroHistory[i], gyroHistory[i+1], t);
      }
    }
  }
  
  // 绘制曲线
  for (int i = 1; i < (NUM_POINTS - 1) * INTERPOLATION + 1; i++) {
    float x1 = (i-1) * xStep / INTERPOLATION;
    float x2 = i * xStep / INTERPOLATION;
    float y1 = axisY - interpolatedPoints[i-1] * scale;
    float y2 = axisY - interpolatedPoints[i] * scale;
    
    y1 = constrain(y1, axisY - axisLengthY, axisY);
    y2 = constrain(y2, axisY - axisLengthY, axisY);
    
    display.drawLine(x1, y1, x2, y2, SSD1306_WHITE);
  }
}
```

```cpp
#define Y_AXIS_RANGE 40      // Y轴显示范围（总高度40像素）
float gyroHistory[SAMPLE_POINTS] = {0}; // 历史速度数组（环形缓冲区）
int currentIndex = 0;                   // 当前数据索引

// 定义全局变量
int axisY;
int axisLengthY;

void setup() {
  // 初始化I2C总线，指定SDA和SCL引脚
  Wire.begin(21, 22);

  // SSD1306_SWITCHCAPVCC = generate display voltage from 3.3V internally
  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println(F("SSD1306 allocation failed"));
    for (;;);
  }
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
  display.setCursor(0, 0);

  Wire.beginTransmission(MPU_addr);
  Wire.write(0x6B);  // PWR_MGMT_1 register
  Wire.write(0);     // set to zero (wakes up the MPU - 6050)
  Wire.endTransmission(true);
  Serial.begin(115200);
}

void loop() {
  // Read accelerometer data
  readAccelData();
  float angleX = calculateAngle();

  // Read gyroscope data
  readGyroData();
  float gyroYDegPerSec = abs((float)GyY / 131.0); // 转换为正值

  // 更新环形缓冲区
  gyroHistory[currentIndex] = gyroYDegPerSec;
  currentIndex = (currentIndex + 1) % SAMPLE_POINTS;

  char angleStr[10];
  dtostrf(angleX, 6, 2, angleStr); // 将浮点数转换为字符串，保留2位小数

  int stringLength = getStringLength(angleStr);
  int digitWidth = 5;

  int chineseWidth = 16 * 2; // “当”“前”总宽度
  int numberWidth = stringLength * digitWidth;
  int unitWidth = 16; // “°”宽度
  int totalWidth = chineseWidth + numberWidth + unitWidth;
  int startX = (SCREEN_WIDTH - totalWidth) / 2; // 计算起始X坐标

  display.clearDisplay();
  display.drawBitmap(startX, 0, font_dang, 16, 16, SSD1306_WHITE);
  display.drawBitmap(startX + 16, 0, font_qian, 16, 16, SSD1306_WHITE);
  display.setCursor(startX + 16 + 16, 5);
  display.print(angleStr);
  int nextX = startX + 16 + 16 + numberWidth;
  display.drawBitmap(nextX, 0, font_du, 16, 16, SSD1306_WHITE);

  drawCoordinateSystem();
  drawGyroCurve();

  display.display();

  Serial.print("当前角度（");
  Serial.print(angleX);
  Serial.print("）°，y轴旋转速度（");
  Serial.print(gyroYDegPerSec);
  Serial.println("）°/s");
  delay(100);
}

// 读取加速度计数据
void readAccelData() {
  Wire.beginTransmission(MPU_addr);
  Wire.write(0x3B);  // starting with register 0x3B (ACCEL_XOUT_H)
  Wire.endTransmission(false);
  Wire.requestFrom((uint8_t)MPU_addr, 6, true); // request a total of 6 registers (X, Y, Z)

  AcX = Wire.read() << 8 | Wire.read(); // 0x3B (ACCEL_XOUT_H) & 0x3C (ACCEL_XOUT_L)
  Wire.read(); Wire.read(); // skip Y axis
  AcZ = Wire.read() << 8 | Wire.read(); // 0x43 (ACCEL_ZOUT_H) & 0x44 (ACCEL_ZOUT_L)
}

// 读取陀螺仪数据
void readGyroData() {
  Wire.beginTransmission(MPU_addr);
  Wire.write(0x45);  // starting with register 0x45 (GYRO_YOUT_H)
  Wire.endTransmission(false);
  Wire.requestFrom((uint8_t)MPU_addr, 2, true); // request 2 registers for Y axis gyro data

  if (Wire.available() >= 2) {
    GyY = Wire.read() << 8 | Wire.read();
  } else {
    GyY = 0;
  }
}

// 计算角度
float calculateAngle() {
  float accelXG = (float)AcX / 16384.0;
  float accelZG = (float)AcZ / 16384.0;
  float angleX = atan2(accelXG, accelZG) * (180 / PI);

  if (angleX < 0) {
    angleX += 360;
  }
  return angleX;
}

// 绘制坐标轴
void drawCoordinateSystem() {
  int axisX = 0; // 坐标轴起点 X 坐标，位于屏幕最左边
  axisY = SCREEN_HEIGHT - 1; // 坐标轴起点 Y 坐标，位于屏幕最下边
  int axisLengthX = SCREEN_WIDTH - 1; // X 轴长度
  axisLengthY = Y_AXIS_RANGE; // Y 轴长度

  // 绘制 X 轴
  display.drawLine(axisX, axisY, axisX + axisLengthX, axisY, SSD1306_WHITE);
  // 绘制 Y 轴
  display.drawLine(axisX, axisY, axisX, axisY - axisLengthY, SSD1306_WHITE);

  // 绘制 Y 轴箭头
  display.drawLine(axisX, axisY - axisLengthY, axisX - 2, axisY - axisLengthY + 5, SSD1306_WHITE);
  display.drawLine(axisX, axisY - axisLengthY, axisX + 2, axisY - axisLengthY + 5, SSD1306_WHITE);

  // 标注 Y 轴范围
  display.setTextSize(0.7);
  display.setCursor(5, axisY - axisLengthY - 2);
  display.print("200");
  display.setCursor(5, axisY + 2);
  display.print("0");
}

// 绘制陀螺仪速度曲线
void drawGyroCurve() {
  int xOffset = 0;
  int yMid = SCREEN_HEIGHT - 1;

  float scaleFactor = Y_AXIS_RANGE / 200.0;

  for (int i = 0; i < SAMPLE_POINTS - 1; i++) {
    int idx1 = (currentIndex + i) % SAMPLE_POINTS;
    int idx2 = (currentIndex + i + 1) % SAMPLE_POINTS;

    float val1 = gyroHistory[idx1];
    float val2 = gyroHistory[idx2];

    int x1 = xOffset + i;
    int y1 = yMid - (val1 * scaleFactor);
    int x2 = xOffset + i + 1;
    int y2 = yMid - (val2 * scaleFactor);

    y1 = constrain(y1, axisY - axisLengthY, axisY);
    y2 = constrain(y2, axisY - axisLengthY, axisY);

    display.drawLine(x1, y1, x2, y2, SSD1306_WHITE);
  }
}
```
